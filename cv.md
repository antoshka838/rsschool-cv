## Code example 

```javascript
function bubbleSort(arr) {
  const n = arr.length;
  let swapped;

  do {
    swapped = false;
    for (let i = 1; i < n; i++) {
      if (arr[i - 1] > arr[i]) {
        [arr[i - 1], arr[i]] = [arr[i], arr[i - 1]];
        swapped = true;
      }
    }
  } while (swapped);

  return arr;
}

const numbers = [5, 2, 9, 1, 7];
console.log(bubbleSort(numbers)); // [1, 2, 5, 7, 9]
```
